üì¶ Installation & Usage

Get started in seconds

NPM Installation

npm install saas-effects

React Usage

import { BackgroundGradient, CardWithCornerAccent } from 'saas-effects';
import 'saas-effects/dist/saas-effects.css';

// Use presets
<BackgroundGradient preset="skyPink" blur="soft">
  <h1>Hello World</h1>
</BackgroundGradient>

// Or custom colors
<BackgroundGradient
  colorStart="#667eea"
  colorEnd="#764ba2"
  angle={90}
  blur="normal"
>
  <h1>Custom Gradient</h1>
</BackgroundGradient>
